%titlepage
%argument is path to assets folder
\newcommand{\createtitle}[1]
{
	\afterpage{
	\begin{textblock*}{8.5in}[0.5,0.5](5.9in,4in)
		\includegraphics[height=1.25in]{#1lasalogo_white}
	\end{textblock*}
	\begin{textblock*}{7.5in}[0,0](0.5in,5.5in)
		\centering
		{\color{white}
		{\titlefont
			\chaptertitle\\
		}
		\vspace{10 mm}
		{\datefont
			\today\\
		}}
	\end{textblock*}
	
	\begin{textblock*}{7.5in}[0,0](0.5in,10in)
		\centering
		{\color{white}
		{\subfont
			\sectiontitle
		}}
	\end{textblock*}
	\pagecolor{purple}
	\fancyfoot[]{}
	\fancyhead{}
	\color{black}
	\newpage}
	\maketitle
	
	%second blank page
	\pagecolor{white}
	\blankpage
	
	\setupheaders{#1}
}

			}
			\vspace{10 mm}
			{\datefont
				\today\\
			}}
		\end{textblock*}
		
		\begin{textblock*}{7.5in}[0,0](0.5in,10in)
			\centering
			{\color{white}
				{\subfont
					\sectiontitle
				}}
			\end{textblock*}
			\pagecolor{purple}
			\fancyfoot[]{}
			\fancyhead{}
			\color{black}
			\newpage
		
      %second blank page
      \blankpage

      \setupheaders{#1}
}
